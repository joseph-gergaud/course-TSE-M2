function backtracking(g,fₖ,∇fₖ,dₖ;α₀=1.,ρ=0.8,c₁=1.e-4)
    αₖ = α₀

    while g(αₖ) > fₖ + c₁αₖ∇fₖ'*dₖ
        αₖ = ρ*αₖ
    return α
end
